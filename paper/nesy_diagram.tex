\begin{figure}[t]
\centering
\begin{tikzpicture}[
    node distance=0.6cm and 1.2cm,
    box/.style={rectangle, draw, rounded corners=3pt, minimum height=0.9cm,
                minimum width=2.4cm, align=center, font=\small},
    input/.style={box, fill=gray!15},
    symbolic/.style={box, fill=orange!20},
    neural/.style={box, fill=blue!15},
    output/.style={box, fill=green!15},
    arrow/.style={-{Stealth[length=2.5mm]}, thick},
    label/.style={font=\scriptsize\itshape, text=gray!70!black},
]

% Input
\node[input] (act) {Notarial Act\\(raw text)};

% Two parallel paths
% Left: Symbolic
\node[symbolic, below left=1.0cm and 0.8cm of act] (norm) {Unicode NFD\\normalisation};
\node[symbolic, below=0.6cm of norm] (rules) {Rule engine\\($\sim$70 rules)};
\node[symbolic, below=0.6cm of rules] (evidence) {Evidence summary\\genre, confidence,\\matched phrases};

% Right: Context
\node[input, below right=1.0cm and 0.8cm of act] (taxonomy) {Genre taxonomy\\(192 lines)};

% Merge
\node[neural, below=1.8cm of act, minimum width=3.6cm, yshift=-3.2cm] (prompt) {Prompt assembly};

% LLM
\node[neural, below=0.6cm of prompt, minimum width=3.6cm] (llm) {LLM\\(12 models)};

% Output
\node[output, below=0.6cm of llm, minimum width=3.6cm] (json) {JSON output\\\texttt{\{core, ext, hybrid\}}};

% Arrows
\draw[arrow] (act) -- (norm);
\draw[arrow] (act) -- (taxonomy);
\draw[arrow] (norm) -- (rules);
\draw[arrow] (rules) -- (evidence);
\draw[arrow] (evidence) -- (prompt);
\draw[arrow] (taxonomy) -- (prompt);
\draw[arrow] (act) -- (prompt);
\draw[arrow] (prompt) -- (llm);
\draw[arrow] (llm) -- (json);

% Labels
\node[label, left=0.15cm of norm] {local};
\node[label, left=0.15cm of rules] {deterministic};
\node[label, right=0.15cm of llm] {API call};

% Background boxes
\begin{scope}[on background layer]
    \node[fill=orange!8, rounded corners=5pt,
          fit=(norm)(rules)(evidence),
          inner sep=0.25cm, label={[font=\small\bfseries]above:Symbolic}] {};
    \node[fill=blue!8, rounded corners=5pt,
          fit=(prompt)(llm),
          inner sep=0.25cm, label={[font=\small\bfseries]above right:Neural}] {};
\end{scope}

\end{tikzpicture}
\caption{NeSy pipeline architecture. The symbolic engine (left, orange) runs locally and produces a structured evidence summary. This is combined with the act text and taxonomy into a prompt for the LLM (blue). The model outputs a 3-tier JSON classification.}
\label{fig:nesy-pipeline}
\end{figure}
